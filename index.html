function convertFormulaToChords(formula, key) {
  // Define scale degrees for each key
  const scaleDegrees = {
    'C': ['C', 'D', 'E', 'F', 'G', 'A', 'B'],
    'Db': ['Db', 'Eb', 'F', 'Gb', 'Ab', 'Bb', 'C'],
    'D': ['D', 'E', 'F#', 'G', 'A', 'B', 'C#'],
    'Eb': ['Eb', 'F', 'G', 'Ab', 'Bb', 'C', 'D'],
    'E': ['E', 'F#', 'G#', 'A', 'B', 'C#', 'D#'],
    'F': ['F', 'G', 'A', 'Bb', 'C', 'D', 'E'],
    'Gb': ['Gb', 'Ab', 'Bb', 'Cb', 'Db', 'Eb', 'F'],
    'G': ['G', 'A', 'B', 'C', 'D', 'E', 'F#'],
    'Ab': ['Ab', 'Bb', 'C', 'Db', 'Eb', 'F', 'G'],
    'A': ['A', 'B', 'C#', 'D', 'E', 'F#', 'G#'],
    'Bb': ['Bb', 'C', 'D', 'Eb', 'F', 'G', 'A'],
    'B': ['B', 'C#', 'D#', 'E', 'F#', 'G#', 'A#']
  };
  
  // Special case for minor keys
  if (key.includes('m')) {
    // Handle minor keys by finding relative major
    // For now, we'll just return an empty array as placeholder
    return [];
  }
  
  // Make sure we have the scale degrees for this key
  if (!scaleDegrees[key]) {
    console.error(`Scale degrees not defined for key: ${key}`);
    return [];
  }
  
  return formula.map(degreeSymbol => {
    // Parse the Roman numeral and chord quality
    const romanNumeralMatch = degreeSymbol.match(/^(b?)([ivIV]+)([^0-9]*[0-9]*)$/);
    if (!romanNumeralMatch) return "";
    
    const [_, flat, roman, quality] = romanNumeralMatch;
    
    // Convert Roman numeral to index
    let degreeIndex;
    const romanLower = roman.toLowerCase();
    if (romanLower === 'i') degreeIndex = 0;
    else if (romanLower === 'ii') degreeIndex = 1;
    else if (romanLower === 'iii') degreeIndex = 2;
    else if (romanLower === 'iv') degreeIndex = 3;
    else if (romanLower === 'v') degreeIndex = 4;
    else if (romanLower === 'vi') degreeIndex = 5;
    else if (romanLower === 'vii') degreeIndex = 6;
    
    // Get the root note
    let rootNote = scaleDegrees[key][degreeIndex];
    
    // Apply flat if needed
    if (flat === 'b') {
      const semitone = noteToSemitone[rootNote];
      const flatSemitone = (semitone - 1 + 12) % 12;
      // Find note name from semitone
      for (const [note, semi] of Object.entries(noteToSemitone)) {
        if (semi === flatSemitone && note.length <= 2) {
          rootNote = note;
          break;
        }
      }
    }
    
    // Adjust quality based on Roman numeral case
    let adjustedQuality = quality;
    if (roman === roman.toLowerCase() && !quality.includes('m') && !quality.includes('dim')) {
      // If Roman is lowercase but quality doesn't specify minor, add 'm'
      adjustedQuality = 'm' + quality;
    }
    
    return rootNote + adjustedQuality;
  });
}

// Play a chord using proper 1-3-5 structure
function playChord(chordName) {
  if (!document.getElementById('enable-chords').checked) return;
  
  // Parse the chord to get root and quality
  const { root, quality } = parseChord(chordName);
  
  // Define intervals for different chord types
  const intervals = {
    "": [0, 4, 7],              // Major (1-3-5)
    "m": [0, 3, 7],             // Minor (1-b3-5)
    "7": [0, 4, 7, 10],         // Dominant 7th (1-3-5-b7)
    "m7": [0, 3, 7, 10],        // Minor 7th (1-b3-5-b7)
    "maj7": [0, 4, 7, 11],      // Major 7th (1-3-5-7)
    "dim": [0, 3, 6],           // Diminished (1-b3-b5)
    "aug": [0, 4, 8],           // Augmented (1-3-#5)
    "m7b5": [0, 3, 6, 10],      // Half-diminished (1-b3-b5-b7)
    "7b9": [0, 4, 7, 10, 13]    // Dominant 7 flat 9 (1-3-5-b7-b9)
  };
  
  // Get the appropriate intervals
  const chordIntervals = intervals[quality] || intervals[""];
  
  // Calculate the notes to play
  const notes = getChordNotes(root, chordIntervals);
  
  // Set volume based on user setting
  const volume = document.getElementById('chord-volume').value / 100;
  
  // Play each note
  notes.forEach(note => {
    playNote(note, volume);
  });
}

// Parse a chord name into root and quality
function parseChord(chordName) {
  let root = chordName.charAt(0);
  let i = 1;
  
  // Handle accidentals (sharp/flat)
  if (i < chordName.length && (chordName.charAt(i) === '#' || chordName.charAt(i) === 'b')) {
    root += chordName.charAt(i);
    i++;
  }
  
  // Extract quality (everything after the root)
  const quality = chordName.substring(i);
  
  return { root, quality };
}

// Get notes for a chord based on root and intervals
function getChordNotes(root, intervals) {
  const rootSemitone = noteToSemitone[root];
  if (rootSemitone === undefined) return [];
  
  // Calculate the semitones for each note in the chord
  return intervals.map(interval => {
    const noteSemitone = (rootSemitone + interval) % 12;
    // Find note name from semitone
    for (const [note, semitone] of Object.entries(noteToSemitone)) {
      if (semitone === noteSemitone && note.length <= 2) {
        return note;
      }
    }
    return "";
  }).filter(note => note); // Remove any empty notes
}

// Play an individual note
function playNote(note, volume = 1.0) {
  // Get the audio element for this note
  let noteAudio = document.getElementById(note);
  
  // Try alternate notation if not found (e.g., Ab might be stored as G#)
  if (!noteAudio && enharmonicMap[note]) {
    noteAudio = document.getElementById(enharmonicMap[note]);
  } else if (!noteAudio && reverseEnharmonicMap[note]) {
    noteAudio = document.getElementById(reverseEnharmonicMap[note]);
  }
  
  if (noteAudio) {
    noteAudio.volume = volume;
    noteAudio.currentTime = 0; // Reset to start
    noteAudio.play();
  }
}

// Play a chord using proper 1-3-5 structure
function playChord(chordName) {
  if (!document.getElementById('enable-chords').checked) return;
  
  // Parse the chord to get root and quality
  const { root, quality } = parseChord(chordName);
  
  // Get the appropriate intervals
  const chordIntervals = chordTypeIntervals[quality] || chordTypeIntervals[""];
  
  // Calculate the notes to play
  const notes = getChordNotes(root, chordIntervals);
  
  // Set volume based on user setting
  const volume = document.getElementById('chord-volume').value / 100;
  
  // Play each note
  notes.forEach(note => {
    playNote(note, volume);
  });
}

// Parse a chord name into root and quality
function parseChord(chordName) {
  let root = chordName.charAt(0);
  let i = 1;
  
  // Handle accidentals (sharp/flat)
  if (i < chordName.length && (chordName.charAt(i) === '#' || chordName.charAt(i) === 'b')) {
    root += chordName.charAt(i);
    i++;
  }
  
  // Extract quality (everything after the root)
  const quality = chordName.substring(i);
  
  return { root, quality };
}

// Get notes for a chord based on root and intervals
function getChordNotes(root, intervals) {
  const rootSemitone = noteToSemitone[root];
  if (rootSemitone === undefined) return [];
  
  // Calculate the semitones for each note in the chord
  return intervals.map(interval => {
    const noteSemitone = (rootSemitone + interval) % 12;
    // Find note name from semitone
    for (const [note, semitone] of Object.entries(noteToSemitone)) {
      if (semitone === noteSemitone && note.length <= 2) {
        return note;
      }
    }
    return "";
  }).filter(note => note); // Remove any empty notes
}

// Play an individual note
function playNote(note, volume = 1.0) {
  // Get the audio element for this note
  let noteAudio = document.getElementById(note);
  
  // Try alternate notation if not found (e.g., Ab might be stored as G#)
  if (!noteAudio && enharmonicMap[note]) {
    noteAudio = document.getElementById(enharmonicMap[note]);
  } else if (!noteAudio && reverseEnharmonicMap[note]) {
    noteAudio = document.getElementById(reverseEnharmonicMap[note]);
  }
  
  if (noteAudio) {
    noteAudio.volume = volume;
    noteAudio.currentTime = 0; // Reset to start
    noteAudio.play();
  }
}
